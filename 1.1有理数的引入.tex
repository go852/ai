% 有理数的引入
\begin{frame}{1.1 有理数的引入}
\textbf{正整数、0 和负整数统称为整数(integer), 正分数和负分数统称为分数
(fraction).\\
整数和分数统称为有理数(rational number).}
\vspace{24pt}
\begin{columns}
\column{0.5\textwidth}
\[
\mbox{有理数}\begin{cases}
\mbox{整数} \begin{cases}
    \mbox{正整数} \\
    0 \\
    \mbox{负整数}
    \end{cases} \\
\mbox{分数}  \begin{cases}
    \mbox{正分数} \\
    \mbox{负分数}
    \end{cases}
\end{cases}
\]

\column{0.5\textwidth}
\[
\mbox{小数}\begin{cases}
\mbox{有限小数} \\
\mbox{无限小数} \begin{cases} 
\mbox{无限循环} \\
\mbox{无限不循环小数}
\end{cases}
\end{cases}
\]
